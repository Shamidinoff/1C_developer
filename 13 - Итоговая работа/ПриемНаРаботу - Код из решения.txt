
Процедура Печать(ТабДок, Ссылка) Экспорт
	//{{_КОНСТРУКТОР_ПЕЧАТИ(Печать)
	ТабДок = Новый ТабличныйДокумент;
	Макет = Документы.ПриемНаРаботу.ПолучитьМакет("Печать");
	Запрос = Новый Запрос;
	Запрос.Текст =
	"ВЫБРАТЬ
	|	ПриемНаРаботу.Номер КАК Номер,
	|	ПриемНаРаботу.Дата КАК Дата,
	|	ПриемНаРаботу.ДатаПриема КАК ДатаПриема,
	|	ПриемНаРаботу.ИспСрок КАК ИспСрок,
	|	ПриемНаРаботу.УсловиеПриема КАК УсловиеПриема,
	|	ПриемНаРаботу.Оклад КАК Оклад,
	|	ПриемНаРаботу.Подразделение КАК Подразделение,
	|	ПриемНаРаботу.Сотрудник КАК Сотрудник,
	|	ПриемНаРаботу.Должность КАК Должность,
	|	НазваниеОрганизация.Значение КАК Организация,
	|	ГенеральныйДиректор.Значение КАК Директор
	|ИЗ
	|	Документ.ПриемНаРаботу КАК ПриемНаРаботу,
	|	Константа.НазваниеОрганизация КАК НазваниеОрганизация,
	|	Константа.ГенеральныйДиректор КАК ГенеральныйДиректор
	|ГДЕ
	|	ПриемНаРаботу.Ссылка В(&СсылкаОбъекта)";
	Запрос.Параметры.Вставить("СсылкаОбъекта", Ссылка);
	Выборка = Запрос.Выполнить().Выбрать();

	ОбластьЗаголовок = Макет.ПолучитьОбласть("Заголовок");
	ОбластьШапка = Макет.ПолучитьОбласть("Шапка");
	ОбластьТело = Макет.ПолучитьОбласть("Тело");
	ОбластьПодвал = Макет.ПолучитьОбласть("Подвал");
	ТабДок.Очистить();

	ВставлятьРазделительСтраниц = Ложь;
	Пока Выборка.Следующий() Цикл
		Если ВставлятьРазделительСтраниц Тогда
			ТабДок.ВывестиГоризонтальныйРазделительСтраниц();
		КонецЕсли;
		
		ОбластьЗаголовок.Параметры.Заполнить(Выборка);
		ТабДок.Вывести(ОбластьЗаголовок, Выборка.Уровень());

		ОбластьШапка.Параметры.Заполнить(Выборка);
		ТабДок.Вывести(ОбластьШапка, Выборка.Уровень());
		
		ОбластьТело.Параметры.Заполнить(Выборка);
		ТабДок.Вывести(ОбластьТело, Выборка.Уровень());
		
		ОбластьПодвал.Параметры.Заполнить(Выборка);
		ТабДок.Вывести(ОбластьПодвал, Выборка.Уровень());

		ВставлятьРазделительСтраниц = Истина;
	КонецЦикла;
	//}}
КонецПроцедуры
